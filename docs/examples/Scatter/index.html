import LayerCake from 'layercake';
import points from './data/points.js';
import Scatter from './components/Scatter.html';
import Voronoi from './components/Voronoi.html';
import AxisX from './components/AxisX.html';
import AxisY from './components/AxisY.html';

const r = 3;
const padding = 4;

const myCake = new LayerCake({
  padding: { top: 0, right: 5, bottom: 20, left: 25 },
  x: 'myX',
  y: 'myY',
  xPadding: [padding, padding],
  yPadding: [padding, padding],
  data: points,
  target: document.getElementById('my-chart'),
  r_: r
})
  .canvasLayers([
    { component: Scatter, opts: { color: '#ff4eff' } }
  ])
  .svgLayers([
    { component: AxisX, opts: {} },
    { component: AxisY, opts: {} },
    { component: Voronoi, opts: {} }
  ]);

myCake.render();
<script>
import { scaleCanvas } from 'layercake';
export default {
  /* --------------------------------------------
   * Put our canvas drawing functions inside of onstate
   * which will change on first load and whenever
   * the store update fires
   */
  onstate () {
    const { canvas, ctx, opts } = this.get();
    const { width, height, xGet, yGet, data, r_ } = this.store.get();

    /* --------------------------------------------
     * If you were to have multiple canvas layers
     * maybe for some artistic layering purposes
     * put these reset functions in the first layer, not each one
     * since they should only run once per update
     */
    scaleCanvas(canvas, ctx, width, height);
    ctx.clearRect(0, 0, width, height);

    /* --------------------------------------------
     * Draw our scatterplot
     */
    data.forEach(d => {
      ctx.beginPath();
      ctx.arc(xGet(d), yGet(d), r_, 0, 2 * Math.PI, false);
      ctx.fillStyle = opts.color;
      ctx.fill();
    });
  }
};
</script>

<script>
import { Delaunay } from 'd3-delaunay';
import * as _ from 'underscore';

export default {
  namespace: 'svg',
  computed: {
    points: ({ $data, $xGet, $yGet }) => {
      const points = _.chain($data)
        .map(d => {
          const point = [$xGet(d), $yGet(d)];
          point.data = d;
          return point;
        })
        .uniq(d => d.join(''))
        .value();
      return points;
    },
    voronoi: ({ $width, $height, $padding, points }) => {
      const delaunay = Delaunay.from(points);
      const voronoi = delaunay.voronoi([0, 0, $width, $height]);
      return voronoi;
    }
  },
  methods: {
    log (e, point) {
      console.log(point, point.data);
    }
  }
};
</script>

<script>
export default {
  namespace: 'svg',
  computed: {
    ticks: ({ $xScale, opts }) => {
      return $xScale.ticks(opts.ticks || opts.tickNumber);
    },
    textAnchor: ({ ticks, opts }) => {
      return function (i) {
        if (opts.snapTicks === true) {
          if (i === 0) {
            return 'start';
          }
          if (i === ticks.length - 1) {
            return 'end';
          }
        }
        return 'middle';
      };
    }
  }
};
</script>
<g class='axis y-axis' transform='translate(-{$padding.left}, 0)'>
